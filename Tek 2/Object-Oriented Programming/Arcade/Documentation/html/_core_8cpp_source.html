76 void *DLLoader::loadLib(const std::string &path) {
77  auto it = _loaded_libs.find(path);
78  if (it != _loaded_libs.end()) {
79  return it->second;
80  }
81  void* handle = dlopen(path.c_str(), RTLD_LAZY);
82  try {
83  if (handle == nullptr)
84  throw DLErrorException();
85  } catch (DLErrorException &e) {
86  std::cerr << e.what() << std::endl;
87  exit(84);
88  }
89  _loaded_libs[path] = handle;

213  else if (input == Input::UP && !isGame)
214  _selectedGraphicsIndex = (_selectedGraphicsIndex + graphics.size() - 1) % graphics.size();
215  else if (input == Input::DOWN && !isGame)

217  else if (input == Input::LEFT)
218  isGame = !isGame;
219  else if (input == Input::RIGHT)
220  isGame = !isGame;
221  else if (input == Input::ENTER)
222  launchGame(isGame, games, graphics);
223  else if (input == Input::A && _gameStarted)
224  switchLibs(graphics, false);
225  else if (input == Input::E && _gameStarted)
226  switchLibs(graphics, true);
